## 三、帧同步状态机（解决粘包/拆包）

### 3.1 问题描述

```
串口/TCP 数据流中常见问题:

粘包: 两个帧收到时连在一起
  发送:  [Frame1][Frame2]
  接收:  [Frame1Frame2] → 一次读到了两个帧

拆包: 一帧被分成多次收到
  发送:  [Frame1]
  接收:  [Frame 前半][Frame1 后半]

解决: 必须用状态机 + 缓冲区
```

### 3.2 完整的状态机解析器

```python
from enum import Enum, auto
from collections import deque
import struct

class ParserState(Enum):
    WAIT_HEADER = auto()  # 等待帧头
    READ_HEADER = auto()  # 读取帧头剩余
    READ_LENGTH = auto()  # 读取长度字段
    READ_PAYLOAD = auto() # 读取载荷
    READ_CHECKSUM = auto()# 读取校验
    DONE = auto()         # 完整帧就绪

class FrameParser:
    """通用帧解析器 - 状态机实现"""

    FRAME_HEADER = b'\xAA\x55'
    HEADER_SIZE = 2
    LENGTH_SIZE = 2      # payload 长度用 2 字节表示
    CHECKSUM_SIZE = 2    # CRC16

    def __init__(self):
        self.state = ParserState.WAIT_HEADER
        self.buffer = bytearray()  # 累积缓冲区
        self.payload_len = 0
        self.frames = deque()      # 解析完成的帧

    def feed(self, data: bytes):
        """喂入新数据"""
        self.buffer.extend(data)

        while True:
            if self.state == ParserState.WAIT_HEADER:
                # 寻找帧头
                idx = self.buffer.find(self.FRAME_HEADER)
                if idx == -1:
                    # 没找到帧头，只保留最后一个字节(可能是帧头的一半)
                    self.buffer = self.buffer[-1:]
                    return
                # 丢弃帧头之前的垃圾数据
                self.buffer = self.buffer[idx:]
                self.state = ParserState.READ_HEADER

            elif self.state == ParserState.READ_HEADER:
                if len(self.buffer) < self.HEADER_SIZE + self.LENGTH_SIZE:
                    return  # 等待更多数据
                # 读取 payload 长度 (小端)
                self.payload_len = struct.unpack(
                    '<H',
                    self.buffer[self.HEADER_SIZE:self.HEADER_SIZE + self.LENGTH_SIZE]
                )[0]
                # 安全检查
                if self.payload_len > 4096:
                    # 长度异常，跳过这个帧头继续找
                    self.buffer = self.buffer[1:]
                    self.state = ParserState.WAIT_HEADER
                    continue
                self.state = ParserState.READ_PAYLOAD

            elif self.state == ParserState.READ_PAYLOAD:
                total_needed = self.HEADER_SIZE + self.LENGTH_SIZE \
                             + self.payload_len + self.CHECKSUM_SIZE
                if len(self.buffer) < total_needed:
                    return  # 等待更多数据

                self.state = ParserState.READ_CHECKSUM

            elif self.state == ParserState.READ_CHECKSUM:
                total_size = self.HEADER_SIZE + self.LENGTH_SIZE \
                           + self.payload_len + self.CHECKSUM_SIZE

                frame_data = bytes(self.buffer[:total_size])

                # 验证 CRC
                crc_offset = total_size - self.CHECKSUM_SIZE
                received_crc = struct.unpack('<H', frame_data[crc_offset:])[0]
                calc_crc = self._calc_crc16(frame_data[:crc_offset])

                if received_crc == calc_crc:
                    self.frames.append(frame_data)  # 校验通过，投递
                else:
                    print(f"CRC 校验失败: 收到 0x{received_crc:04X}, "
                          f"计算 0x{calc_crc:04X}")

                # 消费掉这帧，继续解析后续数据
                self.buffer = self.buffer[total_size:]
                self.state = ParserState.WAIT_HEADER

            else:
                break

    def get_frame(self):
        """取出一帧（非阻塞）"""
        if self.frames:
            return self.frames.popleft()
        return None

    @staticmethod
    def _calc_crc16(data: bytes) -> int:
        """CRC16-Modbus"""
        crc = 0xFFFF
        for byte in data:
            crc ^= byte
            for _ in range(8):
                if crc & 0x0001:
                    crc = (crc >> 1) ^ 0xA001
                else:
                    crc >>= 1
        return crc
```

### 3.3 使用示例

```python
parser = FrameParser()

# 模拟串口接收（拆包场景）
parser.feed(b'\xAA')           # 收到帧头第一个字节
parser.feed(b'\x55\x0C\x00')   # 收到帧头第二个字节 + 长度
parser.feed(b'\x00\x11\x22')   # 收到部分载荷
parser.feed(b'\x33\x44\x55\x66\x77\x88\x99\xAA\xBB\xCC')  # 剩余 + CRC
parser.feed(b'\xAA\x55\x06\x00...') # 下一个帧（粘包）

# 取帧
while (frame := parser.get_frame()):
    print(f"收到帧: {frame.hex()}")
```
